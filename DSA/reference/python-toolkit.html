Gotchas (the live-interview killers)

# 1. Mutable default — evaluated ONCE at def time:
def f(x, acc=[]):        # BUG: shared across calls
def f(x, acc=None): acc = acc if acc is not None else []   # fix

# 2. Aliased rows:
[[0] * 3] * 2            # BUG: two refs, one row
[[0] * 3 for _ in range(2)]                                 # fix

# 3. Shallow copies:
path[:]                  # copies the list, not nested lists
copy.deepcopy(board)     # for nested — O(cells), say so

# 4. Late-binding closures in loops:
fs = [lambda: i for i in range(3)]      # all return 2
fs = [lambda i=i: i for i in range(3)]  # fix: bind at def time

# 5. Integer division on negatives:
-7 // 2 == -4  (floors toward -inf);  int(-7 / 2) == -3 (truncates)

# BFS — Lessons 20/28
q, visited = deque([start]), {start}
while q:
    node = q.popleft()
    for nb in neighbors(node):
        if nb not in visited:
            visited.add(nb); q.append(nb)

# backtracking — Lesson 26
def backtrack(i):
    if done(i): result.append(path[:]); return
    for choice in choices(i):
        path.append(choice)
        backtrack(i + 1)
        path.pop()
